00109   memset(output, 0, sizeof(output));
00110   output = "";

00112   for (int i = 0; i < length; i++) {
00113     output += (number & (1<< (length - 1))) >> (length - 1);

00147 /**
00148  * Control two motors using the ComboDirectMode.  This mode does not allow for fine grained
00149  * speed control.
00150  * @param link the sensor port number
00151  * @param channel the channel of the receiver we wish to communicate with, numbered 0-3
00152  * @param _motorB the command to be sent to Motor B
00153  * @param _motorA the command to be sent to Motor A
00154  */
00155 void PFcomboDirectMode(tSensors link, int channel, eCDMMotorCommand _motorB, eCDMMotorCommand _motorA) {
00156   tByteArray _iBuffer;
00157   tByteArray _oBuffer;
00158 
00159   // Clear the input buffer before we start filling it
00160   memset(_iBuffer, 0, sizeof(tByteArray));
00161   memset(_oBuffer, 0, sizeof(tByteArray));
00162 
00163   // This is the unencoded command for the IR receiver
00164   _iBuffer.arr[0] = (channel << 4) + 1;
00165   _iBuffer.arr[1] = (_motorB << 6) + (_motorA << 4);
00166   _iBuffer.arr[1] += 0xF ^ (_iBuffer.arr[0] >> 4) ^ (_iBuffer.arr[0] & 0xF) ^ (_iBuffer.arr[1] >> 4);
00167 
00168   // Setup the header of the I2C packet
00169   _oBuffer.arr[0] = 16;    // Total msg length
00170   _oBuffer.arr[1] = 0x02;  // I2C device address
00171   _oBuffer.arr[2] = 0x42;  // Internal register
00172 
00173   // Generate the data payload
00174   encodeBuffer(_iBuffer, (tByteArray)_oBuffer);                       // Encode PF command
00175 
00176   // Setup the tail end of the packet
00177   _oBuffer.arr[BUF_HEADSIZE + BUF_DATASIZE] = 11;         // Total IR command length
00178   _oBuffer.arr[BUF_HEADSIZE + BUF_DATASIZE + 1] = 0x02;   // IRLink mode 0x02 is PF motor
00179   _oBuffer.arr[BUF_HEADSIZE + BUF_DATASIZE + 2] = 0x01;   // Start transmitting
00180 
00181   transmitIR(link, _oBuffer, channel, 4);
00182 }

00228 /**
00229  * Encode the input buffer into a special format for the IRLink.
00230  *
00231  * Note: this is an internal function and should not be called directly.
00232  * @param iBuffer the data that is be encoded
00233  * @param oBuffer output buffer for encoded data
00234  */
00235 void encodeBuffer(tByteArray &iBuffer, tByteArray &oBuffer) {
00236   int _oByteIdx = 0;
00237   int _oBitIdx = 0;
00238   int _iIndex = 0;              // _iBUffer bit index
00239   int _oIndex = 0;              // _oBuffer bit index
00240   int _len = 0;
00241 
00242   // Calculate the size of the output bit index
00243   _oIndex = (8 * (MAX_ARR_SIZE - BUF_HEADSIZE)) - 1;
00244 
00245   // Start bit is a special case and is encoded as 0x80
00246   oBuffer.arr[START_DATA] = 0x80; // Start bit
00247   _oIndex -= 8;                   // move the index along 8 bits.
00248 
00249   // Bits in the input buffer are encoded as follows:
00250   // 1 is encoded as 10000
00251   // 0 is encoded as 100
00252   // The encoded bits are tacked onto the end of the output
00253   // buffer, byte boundaries are ignored.
00254   for (_iIndex = 0; _iIndex < (2 * 8); _iIndex++) {
00255     _len = (iBuffer.arr[_iIndex / 8] & 0x80) ? 5 : 3;
00256     _oByteIdx = (MAX_ARR_SIZE - 1) - (_oIndex / 8);
00257     _oBitIdx = _oIndex % 8;
00258     oBuffer.arr[_oByteIdx] += (1 << _oBitIdx);
00259     _oIndex -= _len;
00260     iBuffer.arr[_iIndex / 8] <<= 1;
00261   }
00262 
00263   // Finally, add the stop byte to the end of our command
00264   _oByteIdx = (MAX_ARR_SIZE - 1) - (_oIndex / 8);
00265   _oBitIdx = _oIndex % 8;
00266   oBuffer.arr[_oByteIdx] += (1 << _oBitIdx);
00267 
00268 }
