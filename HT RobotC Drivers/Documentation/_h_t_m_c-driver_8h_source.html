00072 /**
00073  * Start the calibration. The sensor should be rotated a little more than 360 along the
00074  * horizontal plane in no less than 20 seconds.  After the sensor has been rotated,
00075  * call HTMCstopCal() to set the sensor back in measurement mode and save the
00076  * calibration data.  This calibration data is stored in the sensor until the
00077  * next calibration.
00078  * @param link the HTMC port number
00079  * @return true if no error occured, false if it did
00080  */
00081 bool HTMCstartCal(tSensors link) {
00082   memset(HTMC_I2CRequest, 0, sizeof(tByteArray));
00083 
00084   HTMC_I2CRequest.arr[0] = 3;                   // Number of bytes in I2C command
00085   HTMC_I2CRequest.arr[1] = HTMC_I2C_ADDR;       // I2C address of compass sensor
00086   HTMC_I2CRequest.arr[2] = HTMC_MODE;           // Set write address to sensor mode register
00087   HTMC_I2CRequest.arr[3] = HTMC_CALIBRATE_CMD;  // The calibration mode command
00088 
00089   // Start the calibration
00090   return writeI2C(link, HTMC_I2CRequest, 0);
00091 }

00094 /**
00095  * Stop the calibration. This should be called no less than 20 seconds after
00096  * HTMCstartCal() and only if the sensor has been rotated more than 360 degrees
00097  * @param link the HTMC port number
00098  * @return true if no error occured, false if it did
00099  */
00100 bool HTMCstopCal(tSensors link) {
00101   memset(HTMC_I2CRequest, 0, sizeof(tByteArray));
00102 
00103   HTMC_I2CRequest.arr[0] = 3;                 // Number of bytes in I2C command
00104   HTMC_I2CRequest.arr[1] = HTMC_I2C_ADDR;     // I2C address of compass sensor
00105   HTMC_I2CRequest.arr[2] = HTMC_MODE;         // Set write address to sensor mode register
00106   HTMC_I2CRequest.arr[3] = HTMC_MEASURE_CMD;  // The measurement mode command
00107 
00108   // Stop the calibration by setting the mode register back to measurement.
00109   if (!writeI2C(link, HTMC_I2CRequest, 1))
00110     return false;
00111 
00112   // Read back the register value to check if an error has occurred.
00113   if (!readI2C(link, HTMC_I2CReply, 1))
00114     return false;
00115 
00116   // The register is equal to 2 if the calibration has failed.
00117   if (HTMC_I2CReply.arr[0] == 2)
00118     return false;
00119 
00120   return true;
00121 }

00124 /**
00125  * Return the current absolute heading
00126  * @param link the HTMC port number
00127  * @return heading in degrees (0 - 359) or -1 if an error occurred.
00128  */
00129 int HTMCreadHeading(tSensors link) {
00130   memset(HTMC_I2CRequest, 0, sizeof(tByteArray));
00131 
00132   HTMC_I2CRequest.arr[0] = 2;               // Number of bytes in I2C command
00133   HTMC_I2CRequest.arr[1] = HTMC_I2C_ADDR;   // I2C address of compass sensor
00134   HTMC_I2CRequest.arr[2] = HTMC_HEAD_U;     // Set write address to sensor mode register
00135 
00136   if (!writeI2C(link, HTMC_I2CRequest, 2))
00137     return -1;
00138 
00139   if (!readI2C(link, HTMC_I2CReply, 2))
00140     return -1;
00141 
00142   // Result is made up of two bytes.  Reassemble for final heading.
00143   return HTMC_I2CReply.arr[0] * 2 + HTMC_I2CReply.arr[1];
00144 }
